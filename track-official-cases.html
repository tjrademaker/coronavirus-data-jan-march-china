    return a*t**2+b*t+c

def r_squared(df,popt,var):
    ss_res = np.sum((df[var] - quadratic_fit(df.index,popt[0],popt[1],popt[2]))**2)
    ss_tot=np.sum((df[var]-np.mean(df[var]))**2)
    return 1-ss_res/ss_tot

def process_df(df):
    
    df["days"]=(df.datetime-df.datetime.min()).dt.days+3
    df=df[["confirmed_cases","deaths","days"]].groupby("days").min()

    for day,(cases,deaths) in enumerate(zip([310,574,896],[6,17,25])):
        new=pd.DataFrame(np.array([cases,deaths]).reshape(-1,2),index=[day],columns=["confirmed_cases","deaths"])
        df=pd.concat([df,new])#,axis=1)

    df.index.name="days"
    df=df.drop_duplicates()
    return df.sort_index()
